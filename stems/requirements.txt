#
# Die Torch-Stacks (torch/torchaudio) werden bewusst im Dockerfile installiert
# (cu118-Index-URL), damit die CUDA-Variante zur Basis passt. Sie stehen hier
# zusaetzlich gepinnt, damit pip die Versionen NICHT durch Transitiv-Deps
# wieder hochzieht (z.B. via pyannote.audio / pytorch-lightning).
#
# ---------------------------------------------------------------------------
# WICHTIG / Begruendung der Pins (Versionskonflikt-Fix):
#   Fehler war: "module 'torchaudio' has no attribute 'AudioMetaData'"
#   Ursache:    torchaudio >=2.8 hat das Top-Level torchaudio.AudioMetaData und
#               torchaudio.list_audio_backends() deprecatet (entfernt in 2.9),
#               waehrend pyannote.audio 3.x diese APIs noch nutzt.
#   Loesung (Option B): torch/torchaudio auf ein konsistentes cu118-Paar 2.1.2
#               heben. torchaudio 2.1.2 besitzt AudioMetaData UND list_audio_backends
#               UND die backend=-Signatur von torchaudio.info (erst ab 2.1 da) ->
#               pyannote.audio 3.3.2 importiert/laeuft sauber.
#   Hinweis:    ctranslate2 muss auf cuDNN-8/CUDA-11.8 bleiben -> 3.24.0 ist die
#               letzte cuDNN-8-kompatible Version (4.x braucht cuDNN 9 / CUDA 12).
# ---------------------------------------------------------------------------

# --- Torch-Stack: muss zum cu118-Wheel-Paar aus dem Dockerfile passen ---
# (torch & torchaudio in gleicher Minor-Version; cu118-Wheels offiziell verfuegbar)
torch==2.1.2
torchaudio==2.1.2

# --- RunPod Serverless SDK ---
runpod>=1.6.0

# --- Forced-Alignment / ASR (cuDNN-8 / CUDA-11.8-Stack) ---
# WICHTIG: faster-whisper >=1.1 verlangt ctranslate2 >=4.0 (cuDNN 9 / CUDA 12) und
# kollidiert daher mit dieser CUDA-11.8/cuDNN-8-Basis. Deshalb der konsistente,
# per pip-Resolver verifizierte Stack:
#   whisperx 3.1.5 -> pinnt faster-whisper==1.0.1 (vertraegt ctranslate2 3.x)
#                  -> pinnt pyannote.audio==3.1.1 (nutzt torchaudio.AudioMetaData,
#                     vorhanden in torchaudio 2.1.2)
# ctranslate2 3.24.0 = letzte cuDNN-8/CUDA-11.8-kompatible Version.
whisperx==3.1.5
pyannote.audio==3.1.1
faster-whisper==1.0.1
ctranslate2==3.24.0

# --- I/O & Audio-Utilities ---
requests>=2.31.0
numpy>=1.24,<2.0          # <2.0 wegen Kompatibilitaet mit torch 2.1.x / ctranslate2 3.24
soundfile>=0.12.1
ffmpeg-python>=0.2.0      # optional; im Handler wird ffmpeg direkt via subprocess genutzt
